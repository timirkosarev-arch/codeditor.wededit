<script type="module">
import { firebaseConfig } from "./config.js";
import { initializeApp } from "https://www.gstatic.com/firebasejs/10.12.2/firebase-app.js";
import { getFirestore, doc, getDoc } from "https://www.gstatic.com/firebasejs/10.12.2/firebase-firestore.js";

const app = initializeApp(firebaseConfig);
const db = getFirestore(app);

const stateLoading = document.getElementById("state-loading");
const stateNotFound = document.getElementById("state-notfound");
const frame = document.getElementById("site-frame");
const projNameEl = document.getElementById("proj-name");

function getSlugFromUrl() {
  // Поддерживает и красивый путь /site/slug (через vercel.json rewrite),
  // и прямой доступ site.html?slug=slug (для локальной проверки).
  const params = new URLSearchParams(window.location.search);
  if (params.get("slug")) return params.get("slug");
  const parts = window.location.pathname.split("/").filter(Boolean);
  return parts[parts.length - 1] || "";
}

function buildDoc(html, css, js) {
  return `<!DOCTYPE html><html><head><meta charset="UTF-8"><style>${css}</style></head><body>${html}
<script>
try {
${js}
} catch (e) { console.error(e); }
<\/script>
</body></html>`;
}

async function main() {
  const slug = getSlugFromUrl();
  if (!slug) {
    stateLoading.classList.add("hidden");
    stateNotFound.classList.remove("hidden");
    return;
  }
  try {
    const snap = await getDoc(doc(db, "publicSites", slug));
    if (!snap.exists()) {
      stateLoading.classList.add("hidden");
      stateNotFound.classList.remove("hidden");
      return;
    }
    const data = snap.data();
    projNameEl.textContent = data.projectName || "";
    frame.srcdoc = buildDoc(data.html || "", data.css || "", data.js || "");
    frame.style.display = "block";
    stateLoading.classList.add("hidden");
  } catch (err) {
    console.error(err);
    stateLoading.classList.add("hidden");
    stateNotFound.classList.remove("hidden");
  }
}
main();
</script>
